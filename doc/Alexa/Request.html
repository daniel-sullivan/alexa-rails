def slots
  @_slots ||= begin
                if intent_request?
                  return [] if help_request?
                  return [] if cancel_request?

                  params["request"]["intent"]["slots"]
                    .inject(HashWithIndifferentAccess.new) do |hash, slot|
                    name = slot[0]
                    data = slot[1]
                    hash[name] = Alexa::Slot.new(data)
                    hash
                  end
                else
                  []
                end
              end
end
